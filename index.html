WindowJS

Vanilla JavaScript plugin for creating complex windows with a minimum of code.

Getting started

To start using this plugin, you just need to download it and include three files.

HTML

...
<head>
	...
	<link rel="stylesheet" type="text/css" href="css/window.css" />
   	<link rel="stylesheet" type="text/css" href="css/fontello.css" />
   	...
</head>

<body>
	...
	<script type="text/javascript" src="js/window.js"></script>
</body>

JavaScript

var w1 = w2 = w3 = w4 = w5 = null;

document.getElementById('open1').addEventListener('click', function(){
	var options = {
		title: 'Window with Ajax content inside !',
		ajaxContent: {
			url: './simple.html',
			method: 'get'
		},
		onClose: function(){
			w1 = null;
		}
	};
	
	w1 = w1 instanceof WindowJS ? w1 : new WindowJS(options);
});


document.getElementById('open2').addEventListener('click', function(){
	var options = {
		width: 700,
		height: 270,
		title: 'Not resizable window !',
		resizable: false,
		content: '<h1>Not resizable window with custom size (width/height)</h1><p>Just an other paragraph.</p>',
		onClose: function(){
			w2 = null;
		}
	};
	
	w2 = w2 instanceof WindowJS ? w2 : new WindowJS(options);
});


document.getElementById('open3').addEventListener('click', function(){
	var options = {
		title: 'Default window !',
		content: '<h1>Default Window</h1><p>With two custom options ("title" and "content")</p>',
		onClose: function(){
			w3 = null;
		}
	};
	
	w3 = w3 instanceof WindowJS ? w3 : new WindowJS(options);
});

document.getElementById('open4').addEventListener('click', function(){
	var options = {
		title: 'Modal window !',
		modal: true,
		content: '<h1>Modal window</h1><p>The content of the 4th window</p><p>And an other paragraph :)</p>',
		onClose: function(){
			w4 = null;
		}
	};
	
	w4 = w4 instanceof WindowJS ? w4 : new WindowJS(options);
});

document.getElementById('open5').addEventListener('click', function(){
	var options = {
		width: 940,
		height: 715,
		title: 'Wikipedia !',
		iframe: 'https://fr.wikipedia.org/',
		maximized: false,
		contentPadding: false,
		onClose: function(){
			w5 = null;
		}
	};
	
	w5 = w5 instanceof WindowJS ? w5 : new WindowJS(options);
});

document.getElementById('open6').addEventListener('click', function(){
	new WindowJS();
});
